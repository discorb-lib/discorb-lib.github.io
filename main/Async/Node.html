def description
  @object_name ||= "#{self.class}:0x#{object_id.to_s(16)}#{@transient ? ' transient' : nil}"

  if @annotation
    "#{@object_name} #{@annotation}"
  elsif line = self.backtrace(0, 1)&.first
    "#{@object_name} #{line}"
  else
    @object_name
  end
end
