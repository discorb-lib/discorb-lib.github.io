def follow_to(target, reason: nil)
  Async do
    @client
      .http
      .request(
        Route.new(
          "/channels/#{@id}/followers",
          "//channels/:channel_id/followers",
          :post
        ),
        { webhook_channel_id: target.id },
        audit_log_reason: reason
      )
      .wait
  end
end
