    disabled: data[:disabled]

def to_hash
  if @style == :link
    {
      type: 2,
      label: @label,
      style: STYLES[@style],
      url: @url,
      emoji: @emoji&.to_hash,
      disabled: @disabled
    }
  else
    {
      type: 2,
      label: @label,
      style: STYLES[@style],
      custom_id: @custom_id,
      emoji: @emoji&.to_hash,
      disabled: @disabled
    }
  end
end
