def delete
  Async do
    @client
      .http
      .request(
        Route.new(
          "/guilds/#{@source_guild_id}/templates/#{@code}",
          "//guilds/:guild_id/templates/:code",
          :delete
        )
      )
      .wait
  end
end

def edit(name = nil, description = Discorb::Unset)
  Async do
    payload = {}
    payload[:name] = name if name
    payload[:description] = description if description != Discorb::Unset
    @client
      .http
      .request(
        Route.new(
          "/guilds/#{@source_guild_id}/templates/#{@code}",
          "//guilds/:guild_id/templates/:code",
          :patch
        ),
        payload
      )
      .wait
  end
end

def update
  Async do
    _resp, data =
      @client
        .http
        .request(
          Route.new(
            "/guilds/#{@source_guild_id}/templates/#{@code}",
            "//guilds/:guild_id/templates/:code",
            :put
          )
        )
        .wait
    _set_data(data)
  end
end
