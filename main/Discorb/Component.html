def to_payload(components)
  tmp_components = []
  tmp_row = []
  components.each do |c|
    case c
    when Array
      tmp_components << tmp_row
      tmp_row = []
      tmp_components << c
    when SelectMenu, TextInput
      tmp_components << tmp_row
      tmp_row = []
      tmp_components << [c]
    else
      tmp_row << c
    end
  end
  tmp_components << tmp_row
  tmp_components
    .filter { |c| c.length.positive? }
    .map { |c| { type: 1, components: c.map(&:to_hash) } }
end
