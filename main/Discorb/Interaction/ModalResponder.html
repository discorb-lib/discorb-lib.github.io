def show_modal(title, custom_id, components)
  Async do
    payload = {
      title: title,
      custom_id: custom_id,
      components: Component.to_payload(components)
    }
    @client
      .http
      .request(
        Route.new(
          "/interactions/#{@id}/#{@token}/callback",
          "//interactions/:interaction_id/:token/callback",
          :post
        ),
        { type: 9, data: payload }
      )
      .wait
  end
end
