def delete!(reason: nil)
  Async do
    @client.http.request(
      Route.new("/guilds/#{@guild}/integrations/#{@id}", "//guilds/:guild_id/integrations/:integration_id",
                :delete), {}, audit_log_reason: reason
    ).wait
  end
end
