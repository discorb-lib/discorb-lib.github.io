def slash(command_name, description, options = {}, &block)
  command =
    Discorb::ApplicationCommand::Command::ChatInputCommand.new(
      command_name,
      description,
      options,
      [],
      block,
      1,
      self,
      nil,
      nil
    )
  @commands << command
  @client.callable_commands << command
  command
end
