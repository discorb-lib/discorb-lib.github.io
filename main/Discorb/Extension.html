def events
  return @events if @events

  ret = {}
  self.class.events.each do |event, handlers|
    ret[event] = handlers.map do |handler|
      Discorb::EventHandler.new(
        proc do |*args, **kwargs|
          instance_exec(*args, **kwargs, &handler[2])
        end,
        handler[0],
        handler[1]
      )
    end
  end
  @events = ret
end
